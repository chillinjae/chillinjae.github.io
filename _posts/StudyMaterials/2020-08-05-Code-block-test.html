def single_replace(parse_dict: dict, text: str):
    """Replace the first occurrence expression in text from dict keys.

    :param parse_dict: dictionary that contains the pair of strings
    """
    regex = re.compile("|".join(map(re.escape, parse_dict.keys())))
    count = len(parse_dict)

    return regex.sub(lambda match: parse_dict[match.group(0)], text, count)
